#include <stdio.h>

char* cont_0a100(){
  int aux;
  int contador;

  FILE *p1;

  p1 = fopen("Arquivo_0_a_100.txt", "w");

  if(p1 == NULL){
    return "Erro na criação do arquivo";
  }
  else{
    for(aux=0; aux<=100; aux++)
    fprintf(p1,"%d\n",aux);

    fclose(p1);
    return "Arquivo_0_a_100.txt criado com sucesso!!";
  }
}
